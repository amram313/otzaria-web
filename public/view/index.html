      function normalizeSpacesKeepNewlines(s) {
        const x = String(s || "").replace(/\r\n/g, "\n")
        return x
          .replace(/[ \t]+\n/g, "\n")
          .replace(/\n[ \t]+/g, "\n")
      }

      function splitToParagraphs(text) {
        const t = normalizeSpacesKeepNewlines(text)
        const parts = t.split(/\n\s*\n+/g)
        return parts
          .map(p => p.replace(/\n+/g, "\n").trim())
          .filter(p => p.length > 0)
      }

      function textFromNodeWithBr(node) {
        let out = ""
        function walk(n) {
          if (!n) return
          if (n.nodeType === Node.TEXT_NODE) { out += n.nodeValue || ""; return }
          if (n.nodeType === Node.ELEMENT_NODE) {
            const tag = (n.tagName || "").toUpperCase()
            if (tag === "BR") { out += "\n"; return }
            if (tag === "SCRIPT" || tag === "STYLE") return
            for (const c of Array.from(n.childNodes || [])) walk(c)
          }
        }
        walk(node)
        return out
      }

      function collectDocxBlocksFromRoot(root) {
        const blocks = []
        let buffer = ""

        function flushBuffer() {
          const paras = splitToParagraphs(buffer)
          paras.forEach(p => blocks.push({ type: "p", text: p }))
          buffer = ""
        }

        function addHeading(level, text) {
          const t = String(text || "").replace(/\s+/g, " ").trim()
          if (!t) return
          blocks.push({ type: "h", level, text: t })
        }

        function addParagraph(text) {
          const paras = splitToParagraphs(text)
          paras.forEach(p => blocks.push({ type: "p", text: p }))
        }

        function walk(node) {
          if (!node) return

          if (node.nodeType === Node.TEXT_NODE) {
            buffer += node.nodeValue || ""
            return
          }

          if (node.nodeType !== Node.ELEMENT_NODE) return

          const tag = (node.tagName || "").toUpperCase()

          if (tag === "H1" || tag === "H2" || tag === "H3" || tag === "H4") {
            flushBuffer()
            addHeading(Number(tag.slice(1)), node.textContent || "")
            return
          }

          if (tag === "BR") { buffer += "\n"; return }

          if (tag === "P" || tag === "BLOCKQUOTE" || tag === "PRE") {
            flushBuffer()
            addParagraph(textFromNodeWithBr(node))
            return
          }

          if (tag === "LI") {
            flushBuffer()
            const txt = textFromNodeWithBr(node).replace(/\s+/g, " ").trim()
            if (txt) blocks.push({ type: "p", text: "• " + txt })
            return
          }

          for (const c of Array.from(node.childNodes || [])) walk(c)

          if (tag === "DIV" || tag === "SECTION" || tag === "ARTICLE") buffer += "\n\n"
        }

        walk(root)
        flushBuffer()
        return blocks
      }

      function docxTextRunXml(text, opts) {
        const o = opts || {}
        const b = o.bold ? "" : ""
        const u = o.underline ? "" : ""
        const color = o.color ? "" : ""
        const sz = o.size ? (
          "" +
          ""
        ) : ""
        return (
          "" +
            "" + b + u + color + sz + "" +
            "" + xmlEsc(text) + "" +
          ""
        )
      }

      function docxPPrXml(pStyle, indentLevelTwips) {
        // indent right for RTL
        const ind = indentLevelTwips ? "" : ""
        return (
          "" +
            (pStyle ? "" : "") +
            "" +
            "" +
            ind +
          ""
        )
      }

        // indent per level (Heading1=0, Heading2=360, Heading3=720, Heading4=1080)

      async function exportDocxFullClickableToc() {
        if (!requireJSZip()) return

        const title = getTitleFromDom()
        const outName = fileStem(baseNameFromPath(currentPath)) + ".docx"

        const root = document.getElementById("content")
        let blocks = []
        if (root && root.style.display !== "none") {
          blocks = collectDocxBlocksFromRoot(root)
        } else {
          const paras = splitToParagraphs(currentPlainText || "")
          blocks = paras.map(p => ({ type: "p", text: p }))
        }

        // Build heading bookmarks + toc list
        let bmId = 10
        const tocEntries = [] // {name, text, level}
        const bodyParts = []

        // DOCX manual TOC (clickable)
        bodyParts.push(docxParagraphXml("תוכן עניינים", "Heading1"))
        // list
        let headingIndex = 0
        blocks.forEach((b) => {
          if (b.type === "h") {
            const lvl = clamp(Number(b.level || 1), 1, 4)
            const name = "bm_" + headingIndex++
            tocEntries.push({ name, text: b.text, level: lvl })
          }
        })

        if (tocEntries.length) {
          tocEntries.forEach((t) => {
            bodyParts.push(docxHyperlinkParaXml(t.name, t.text, t.level))
          })
        } else {
          bodyParts.push(docxParagraphXml("(אין כותרות ליצירת תוכן עניינים)", "Normal"))
        }

        bodyParts.push(docxPageBreakXml())

        // Content with Heading styles + bookmarks at each heading
        let tocCursor = 0
        blocks.forEach((b) => {
          if (b.type === "h") {
            const lvl = clamp(Number(b.level || 1), 1, 4)
            const pStyle = "Heading" + lvl

            const entry = tocEntries[tocCursor++]
            const bmName = entry ? entry.name : ("bm_" + (headingIndex++))

            const start = docxBookmarkStartXml(bmId, bmName)
            const end = docxBookmarkEndXml(bmId)
            bmId++

            bodyParts.push(
              "" +
                docxPPrXml(pStyle, 0) +
                start +
                docxTextRunXml(b.text, { bold: true }) +
                end +
              ""
            )
          } else {
            bodyParts.push(docxParagraphXml(b.text, "Normal"))
          }
        })

        const documentXml =
`

  
    ${bodyParts.join("")}
    
  
`

        const zip = new JSZip()
        zip.file("[Content_Types].xml", docxContentTypesXml())
        zip.folder("_rels").file(".rels", docxRootRelsXml())
        zip.folder("word").file("document.xml", documentXml)
        zip.folder("word").file("styles.xml", docxStylesXml())
        zip.folder("word").folder("_rels").file("document.xml.rels", docxDocumentRelsXml())
        zip.folder("docProps").file("core.xml", docxCoreXml(title))
        zip.folder("docProps").file("app.xml", docxAppXml())

        const blob = await zip.generateAsync({
          type: "blob",
          mimeType: "application/vnd.openxmlformats-officedocument.wordprocessingml.document"
        })

        downloadBlob(blob, outName)
      }

      function ensureHeadingIds(html) {
        const doc = new DOMParser().parseFromString(html, "text/html")
        const headings = doc.querySelectorAll("h1,h2,h3,h4")
        headings.forEach((h, i) => { if (!h.id) h.id = "h-" + i })
        return { doc, headings }
      }

      function buildNestedOlFromHeadings(headings, hrefPrefix) {
        const items = Array.from(headings).map((h) => {
          const lvl = clamp(Number(String(h.tagName).slice(1)), 1, 4)
          const id = String(h.id || "").trim()
          const txt = (h.textContent || "").trim() || "כותרת"
          return { lvl, id, txt }
        }).filter(x => x.id)

        if (!items.length) return "
    "

        let out = []
        let level = 0
        let openLi = false

        function openOl(){ out.push("
      ") }
        function closeOl(){ out.push("
    ") }
        function closeLiIfOpen(){ if (openLi) { out.push(""); openLi = false } }

        for (const it of items) {
          if (level === 0) { openOl(); level = 1 }

          while (it.lvl > level) {
            out.push("
      ")
            level++
          }
          while (it.lvl < level) {
            closeLiIfOpen()
            closeOl()
            level--
          }

          closeLiIfOpen()
          out.push(
            "
    1. " +
            xmlEscape(it.txt) +
            ""
          )
          openLi = true
        }

        closeLiIfOpen()
        while (level > 0) { closeOl(); level-- }
        return out.join("")
      }

          p.fs = clamp(p.fs - 1, 12, 34)

          const p = loadPrefs()
          p.fs = clamp(p.fs + 1, 12, 34)
          savePrefs(p)
          applyPrefs(p)
        }

          const p = loadPrefs()
          p.book = !p.book
          savePrefs(p)
          applyPrefs(p)
        }

        document.getElementById("btnToc").onclick = () => openToc()
        document.getElementById("btnCloseToc").onclick = () => closeToc()
        document.getElementById("overlay").onclick = () => closeToc()

        const path = getParam("path")
        const meta = document.getElementById("meta")
        const loading = document.getElementById("loading")
        const content = document.getElementById("content")
        const pre = document.getElementById("pre")
        const err = document.getElementById("err")
        const pdf = document.getElementById("pdf")
        const btnRaw = document.getElementById("btnRaw")

        if (!path){
          loading.style.display = "none"
          err.style.display = "block"
          err.textContent = "Missing ?path="
          return
        }

        currentPath = path
        currentSafeHtml = ""
        currentPlainText = ""
        currentTocItems = []

        meta.textContent = "path: " + path

        btnRaw.onclick = () => {
          const u = "/lib/" + encodePath(path)
          window.open(u, "_blank", "noopener,noreferrer")
        }

        if (isPdf(path)){
          loading.style.display = "none"
          pdf.style.display = "block"
          pdf.src = "/lib/" + encodePath(path)
          showExportButtonsForPdf()
          return
        }

        const url = "/lib/" + encodePath(path)
        const r = await fetch(url, { cache: "no-store" })
        if (!r.ok){
          loading.style.display = "none"
          err.style.display = "block"
          err.textContent = "Fetch failed: " + r.status + "\n" + url
          return
        }

        const text = await r.text()
        loading.style.display = "none"

        if (looksLikeHtml(text)){
          const safe = sanitizeHtml(text)
          if (safe && safe.trim().length > 10){
            content.style.display = "block"
            content.innerHTML = safe
            currentSafeHtml = safe
            buildNav()
            showExportButtonsForHtmlOrText()
          } else {
            pre.style.display = "block"
            pre.textContent = text
            currentPlainText = text
            currentTocItems = []
            showExportButtonsForHtmlOrText()
          }
        } else {
          pre.style.display = "block"
          pre.textContent = text
          currentPlainText = text
          currentTocItems = []
          showExportButtonsForHtmlOrText()
        }
      }

      main().catch((e) => {
        const loading = document.getElementById("loading")
        const err = document.getElementById("err")
        loading.style.display = "none"
        err.style.display = "block"
        err.textContent = String(e && e.message ? e.message : e)
      })
